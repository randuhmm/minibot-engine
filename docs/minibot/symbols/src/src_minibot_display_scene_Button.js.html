1 define(
  2 	[
  3 		'./SceneDisplayObject',
  4 		'minibot/event/MouseEvent'
  5 	],
  6 	function
  7 	(
  8 		SceneDisplayObject,
  9 		MouseEvent
 10 	)
 11 	{
 12 		
 13 		var Button = Class.create(
 14 			SceneDisplayObject,
 15 			/** @lends display.scene.Button# */
 16 			{
 17 				
 18 				upState: null,
 19 				downState: null,
 20 				overState: null,
 21 				currentState: null,
 22 				
 23 				states: null,
 24 				
 25 				isDown: false,
 26 				isOver: false,
 27 				
 28 				mouseMoveCallback: null,
 29 				mouseUpCallback: null,
 30 				
 31 				/**
 32 				 * Description of constructor.
 33 				 * @class Short description of class.
 34 				 * Long Description of class.
 35 				 * @extends display.scene.SceneDisplayObject
 36 				 * @constructs
 37 				 * @param
 38 				 * @param {display.DisplayObject} upState The resource to display when the button is "up".
 39 				 * @param {display.DisplayObject} downState The resource to display when the button is "down".
 40 				 * @param {display.DisplayObject} overState The resource to display when the button is "over".
 41 				 */
 42 				initialize: function($super, upState, downState, overState)
 43 				{
 44 					$super();
 45 					if(upState != undefined) this.upState = upState;
 46 					if(downState != undefined) this.downState = downState;
 47 					if(overState != undefined) this.overState = overState;
 48 					
 49 					this.currentState = this.upState;
 50 					
 51 					this.w = this.upState.w;
 52 					this.h = this.upState.h;
 53 					
 54 					this.states = new Array();
 55 					if(this.upState != null) this.states.push(this.upState);
 56 					if(this.downState != null) this.states.push(this.downState);
 57 					if(this.overState != null) this.states.push(this.overState);
 58 					
 59 					//this.mouseMoveCallback = this.handleMouseMove.bindAsEventListener(this);
 60 					//this.mouseUpCallback = this.handleMouseUp.bindAsEventListener(this);
 61 				},
 62 				
 63 				/** 
 64 				 * Function description.
 65 				 * @access public
 66 				 */
 67 				render: function(dt, x, y)
 68 				{
 69 					this.currentState.render(dt, this.x + x, this.y + y);
 70 				},
 71 				
 72 				/** 
 73 				 * Function description.
 74 				 * @access protected
 75 				 */
 76 				onAddedToScene: function($super)
 77 				{
 78 					$super();
 79 					this.states.each(function(displayObject) {
 80 						displayObject.root = this.root;
 81 						displayObject.parent = this;
 82 						displayObject.scene = this.scene;
 83 						displayObject.onAddedToScene();
 84 					}.bind(this));
 85 				},
 86 				
 87 				/** 
 88 				 * Function description.
 89 				 * @access protected
 90 				 */
 91 				dispatchEvent: function($super, event)
 92 				{
 93 					if(!this.isDown) {
 94 						if(event.type == MouseEvent.MOUSE_DOWN) {
 95 							this.currentState = this.downState;
 96 							this.isDown = true;
 97 							
 98 							if(!this.mouseMoveCallback) {
 99 								this.mouseMoveCallback = this.handleMouseMove.bindAsEventListener(this);
100 								this.root.addEventListener(MouseEvent.MOUSE_MOVE, this.mouseMoveCallback);
101 							}
102 							
103 							if(!this.mouseUpCallback) {
104 								this.mouseUpCallback = this.handleMouseUp.bindAsEventListener(this);
105 								this.root.addEventListener(MouseEvent.MOUSE_UP, this.mouseUpCallback);
106 							}
107 							
108 						} else if(event.type == MouseEvent.MOUSE_MOVE) {
109 							
110 							
111 							
112 							this.currentState = this.overState;
113 							
114 							if(!this.mouseMoveCallback) {
115 								this.mouseMoveCallback = this.handleMouseMove.bindAsEventListener(this);
116 								this.root.addEventListener(MouseEvent.MOUSE_MOVE, this.mouseMoveCallback);
117 							}
118 							
119 						}
120 					} else {
121 						if(event.type == MouseEvent.MOUSE_UP) {
122 							this.isDown = false;
123 							
124 							if(this.mouseMoveCallback) {
125 								this.root.removeEventListener(MouseEvent.MOUSE_MOVE, this.mouseMoveCallback);
126 								this.mouseMoveCallback = null;
127 							}
128 							
129 							if(this.mouseUpCallback) {
130 								this.root.removeEventListener(MouseEvent.MOUSE_UP, this.mouseUpCallback);
131 								this.mouseUpCallback = null;
132 							}
133 							
134 							this.currentState = this.upState;
135 							
136 							this.sendClick.bind(this).defer(event);
137 						}
138 					}
139 					return $super(event);
140 				},
141 				
142 				/** 
143 				 * Function description.
144 				 * @access private
145 				 */
146 				handleMouseMove: function(event)
147 				{
148 					
149 					this.isOver = this.isEventOver(event);
150 					
151 					if(this.isOver && this.isDown) {
152 						this.currentState = this.downState;
153 					} else if(this.isOver) {
154 						this.currentState = this.overState;
155 					} else if(this.isDown) {
156 						this.currentState = this.upState;
157 					} else {
158 						this.currentState = this.upState;
159 						if(this.mouseMoveCallback) {
160 							this.root.removeEventListener(MouseEvent.MOUSE_MOVE, this.mouseMoveCallback);
161 							this.mouseMoveCallback = null;
162 						}
163 					}
164 					
165 				},
166 				
167 				isEventOver: function(event)
168 				{
169 					var sceneX = this.getSceneX();
170 					var sceneY = this.getSceneY();
171 					return (
172 						event.x >= sceneX && 
173 						event.x <= (sceneX + this.w) && 
174 						event.y >= sceneY && 
175 						event.y <= (sceneY + this.h)
176 					);
177 				},
178 				
179 				/** 
180 				 * Function description.
181 				 * @access private
182 				 */
183 				handleMouseUp: function(event)
184 				{
185 					this.currentState = this.upState;
186 					this.isDown = false;
187 					
188 					if(this.mouseMoveCallback) {
189 						this.root.removeEventListener(MouseEvent.MOUSE_MOVE, this.mouseMoveCallback);
190 						this.mouseMoveCallback = null;
191 					}
192 					
193 					if(this.mouseUpCallback) {
194 						this.root.removeEventListener(MouseEvent.MOUSE_UP, this.mouseUpCallback);
195 						this.mouseUpCallback = null;
196 					}
197 				},
198 				
199 				/** 
200 				 * Function description.
201 				 * @access private
202 				 */
203 				sendClick: function(event)
204 				{
205 					var x = event.x;
206 					var y = event.y;
207 					var type = MouseEvent.CLICK;
208 					
209 					var mouseEvent = new MouseEvent(type, false, false, x, y, this);
210 					this.dispatchEvent(mouseEvent);
211 				}
212 				
213 			}
214 		);
215 		
216 		return Button;
217 		
218 	}
219 );
