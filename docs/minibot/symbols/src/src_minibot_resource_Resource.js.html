  7 define(
  8 	[],
  9 	function()
 10 	{
 11  		
 12 		var Resource = Class.create(
 13 			/** @lends resource.Resource# */
 14 			{
 15 				
 16 				/**
 17 				 * Create a new Resource object. 
 18 				 * @class This is the basic Resource class.
 19 				 * It is intended to be used as an Interface, although such types are not
 20 				 * enforced in JavaScript.
 21 				 * @constructs
 22 				 * @param {String} id The id of the Resource.
 23 				 */
 24 				initialize: function(id)
 25 				{
 26 					this.id = id;
 27 				},
 28 				
 29 				/**
 30 				 * The id of the Resource.
 31 				 * @type String
 32 				 */
 33 				id: null,
 34 				
 35 				/**
 36 				 * Value that represents if the resource has been loaded or not.
 37 				 * @private
 38 				 * @type bool
 39 				 */
 40 				loaded: false,
 41 				
 42 				/**
 43 				 * Load the Resource
 44 				 * @param {ResourceManager} manager The ResourceManager instance.
 45 				 * @param {Function} callback The callback function to call once loaded.
 46 				 */
 47 				load: function(manager, callback)
 48 				{
 49 					// Overload this function in the base class
 50 					this.loaded = true;
 51 					callback.defer();
 52 				},
 53 				
 54 				/**
 55 				 * Returns if the Resource has been loaded or not.
 56 				 * @type bool
 57 				 */
 58 				isLoaded: function()
 59 				{
 60 					return this.loaded;
 61 				}
 62 				
 63 			}
 64 		);
 65 		
 66 		return Resource;
 67 		
 68 	}
 69 );
