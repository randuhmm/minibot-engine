1 define(
  2 	[
  3 		'minibot/display/canvas/CanvasDisplayObject',
  4 		'minibot/event/UIEvent'
  5 	],
  6 	function
  7 	(
  8 		CanvasDisplayObject,
  9 		UIEvent
 10 	)
 11 	{
 12 		
 13 		var Button = Class.create(
 14 			CanvasDisplayObject,
 15 			/** @lends display.canvas.Button# */
 16 			{
 17 				
 18 				upState: null,
 19 				downState: null,
 20 				overState: null,
 21 				currentState: null,
 22 				
 23 				states: null,
 24 				
 25 				isDown: false,
 26 				isOver: false,
 27 				
 28 				touchMoveCallback: null,
 29 				touchEndCallback: null,
 30 				
 31 				/**
 32 				 * Description of constructor.
 33 				 * @class Short description of class.
 34 				 * Long Description of class.
 35 				 * @extends display.canvas.CanvasDisplayObject
 36 				 * @constructs
 37 				 * @param 
 38 				 */
 39 				initialize: function($super, upState, downState, overState)
 40 				{
 41 					$super();
 42 					if(upState != undefined) this.upState = upState;
 43 					if(downState != undefined) this.downState = downState;
 44 					if(overState != undefined) this.overState = overState;
 45 					
 46 					this.currentState = this.upState;
 47 					
 48 					this.w = this.upState.w;
 49 					this.h = this.upState.h;
 50 					
 51 					this.states = new Array();
 52 					if(this.upState != null) this.states.push(this.upState);
 53 					if(this.downState != null) this.states.push(this.downState);
 54 					if(this.overState != null) this.states.push(this.overState);
 55 					
 56 					this.touchMoveCallback = this.handleTouchMove.bindAsEventListener(this);
 57 					this.touchEndCallback = this.handleTouchEnd.bindAsEventListener(this);
 58 				},
 59 				
 60 				onAddedToCanvas: function($super)
 61 				{
 62 					$super();
 63 					this.states.each(function(displayObject) {
 64 						displayObject.root = this.root;
 65 						displayObject.parent = this;
 66 						displayObject.onAddedToCanvas();
 67 					}.bind(this));
 68 				},
 69 				
 70 				render: function(dt, context, x, y)
 71 				{
 72 					this.currentState.render(dt, context, this.x + x, this.y + y);
 73 					//context.strokeStyle = '#000000';
 74 					//context.strokeRect(this.x + x, this.y + y, this.w, this.h);
 75 				},
 76 				
 77 				dispatchEvent: function($super, event)
 78 				{
 79 					if(!this.isDown) {
 80 						if(event.type == UIEvent.TOUCH_START) {
 81 							this.currentState = this.downState;
 82 							this.isDown = true;
 83 							this.root.addEventListener(UIEvent.TOUCH_MOVE, this.touchMoveCallback);
 84 							this.root.addEventListener(UIEvent.TOUCH_END, this.touchEndCallback);
 85 						}
 86 						if(event.type == UIEvent.TOUCH_END) return;
 87 					} else {
 88 						if(event.type == UIEvent.TOUCH_END) {
 89 							this.currentState = this.upState;
 90 							this.isDown = false;
 91 							this.root.removeEventListener(UIEvent.TOUCH_MOVE, this.touchMoveCallback);
 92 							this.root.removeEventListener(UIEvent.TOUCH_END, this.touchEndCallback);
 93 						}
 94 					}
 95 					return $super(event);
 96 				},
 97 				
 98 				handleTouchMove: function(event)
 99 				{
100 					var canvasX = this.getCanvasX();
101 					var canvasY = this.getCanvasY();
102 					if(
103 						event.x >= canvasX && 
104 						event.x <= (canvasX + this.w) && 
105 						event.y >= canvasY && 
106 						event.y <= (canvasY + this.h)
107 					) {
108 						this.currentState = this.downState;
109 						this.isOver = true;
110 					} else {
111 						this.currentState = this.upState;
112 						this.isOver = false;
113 					}
114 				},
115 				
116 				handleTouchEnd: function(event)
117 				{
118 					console.log("handleTouchEnd");
119 					this.currentState = this.upState;
120 					this.isDown = false;
121 					this.root.removeEventListener(UIEvent.TOUCH_MOVE, this.touchMoveCallback);
122 					this.root.removeEventListener(UIEvent.TOUCH_END, this.touchEndCallback);
123 				}
124 				
125 			}
126 		)
127 	}
128 );
