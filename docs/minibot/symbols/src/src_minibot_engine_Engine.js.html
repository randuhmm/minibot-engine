  7 define(
  8 	[
  9 		'minibot/event/EventDispatcher'
 10 	],
 11 	function
 12 	(
 13 		EventDispatcher
 14 	)
 15 	{
 16 		
 17 		var Engine = Class.create(
 18 			EventDispatcher,
 19 			/** @lends engine.Engine# */
 20 			{
 21 				
 22 				// List of all systems
 23 				systems: null,
 24 				
 25 				// List of all systems by type
 26 				systemsByType: null,
 27 				
 28 				// List of all objects
 29 				objects: null,
 30 				
 31 				// List of objects by type
 32 				objectsByType: null,
 33 				
 34 				// The primary camera
 35 				camera: null,
 36 				
 37 				// the player
 38 				player: null,
 39 				
 40 				// The resource map
 41 				resources: null,
 42 				
 43 				// The update order
 44 				updateOrder: null,
 45 				
 46 				initialize: function($super)
 47 				{
 48 					$super();
 49 					
 50 					this.resources = {};
 51 					
 52 					this.systems = [];
 53 					this.systemsByType = {};
 54 					
 55 					this.objects = [];
 56 					this.objectsByType = {};
 57 					
 58 				},
 59 				
 60 				// Public Methods -->
 61 				
 62 				// Update/Render Methods -->
 63 				
 64 				update: function(dt)
 65 				{
 66 					
 67 					// Update the Systems in preset order
 68 					for(var s = 0; s < this.updateOrder.length; s++) {
 69 						this.systemsByType[this.updateOrder[s]].update(dt);
 70 					}
 71 					
 72 				},
 73 				
 74 				render: function(dt)
 75 				{
 76 					// Override this in sub class
 77 				},
 78 				
 79 				renderPhysics: function(dt)
 80 				{
 81 					// Override this in sub class
 82 				},
 83 				
 84 				setUpdateOrder: function(updateOrder)
 85 				{
 86 					this.updateOrder = updateOrder;
 87 				},
 88 				
 89 				// <-- Update/Render Methods
 90 				
 91 				// Object/System Methods -->
 92 				
 93 				addSystem: function(sys)
 94 				{
 95 					// Get type
 96 					var type = sys.getType();
 97 					if(this.systemsByType[type] != undefined) {
 98 						// ERROR?
 99 						return;
100 					}
101 					
102 					// Add to systems
103 					this.systems.push(sys);
104 					
105 					// Add to systemsByType
106 					this.systemsByType[type] = sys;
107 					
108 					sys.setEngine(this);
109 					sys.onAddedToEngine();
110 				},
111 				
112 				removeSystem: function()
113 				{
114 					
115 				},
116 				
117 				addObject: function(obj)
118 				{
119 					// Add to objects
120 					this.objects.push(obj);
121 					
122 					// Add to objectsByType
123 					var type = obj.getType();
124 					if(this.objectsByType[type] == undefined) {
125 						this.objectsByType[type] = [];
126 					}
127 					this.objectsByType[type].push(obj);
128 					
129 					// Add to systems if component is available
130 					for(var i = 0; i < this.systems.length; i++) {
131 						this.systems[i].addObject(obj);
132 					}
133 					
134 					obj.setEngine(this);
135 					obj.onAddedToEngine();
136 				},
137 				
138 				removeObject: function()
139 				{
140 					
141 				},
142 				
143 				// <-- Object/System Methods
144 				
145 				// Resource Methods -->
146 				
147 				getResources: function()
148 				{
149 					return this.resources;
150 				},
151 				
152 				getResource: function(type, id)
153 				{
154 					if(this.resources[type] == undefined) return null;
155 					if(this.resources[type][id] == undefined) return null;
156 					return this.resources[type][id];
157 				},
158 				
159 				addResource: function(type, id)
160 				{
161 					if(this.resources[type] == undefined) this.resources[type] = {};
162 					this.resources[type][id] = null;
163 				},
164 				
165 				onResourcesLoaded: function()
166 				{
167 					var i;
168 					for(i = 0; i < this.systems.length; i++) {
169 						this.systems[i].onResourcesLoaded();
170 					}
171 					for(i = 0; i < this.objects.length; i++) {
172 						this.objects[i].onResourcesLoaded();
173 					}
174 				},
175 				
176 				// <-- Resource Methods
177 				
178 				// <-- Public Methods
179 				
180 				
181 				
182 			}
183 		);
184 		
185 		return Engine;
186 	
187 	}
188 );
