  7 define(
  8 	[
  9 		'minibot/resource/Resource'
 10 	],
 11 	function
 12 	(
 13 		Resource
 14 	)
 15 	{
 16 		
 17 		var SpriteResource = Class.create(
 18 			Resource,
 19 			/** @lends resource.SpriteResource# */
 20 			{
 21 				imageId: null,
 22 				
 23 				imageResource: null,
 24 				
 25 				img: null,
 26 				
 27 				x: 0,
 28 				y: 0,
 29 				w: -1,
 30 				h: -1,
 31 				
 32 				/**
 33 				 * Description of constructor.
 34 				 * @class Short description of class.
 35 				 * Long Description of class.
 36 				 * @extends resource.Resource
 37 				 * @constructs
 38 				 * @param {String} id The id of the Resource.
 39 				 * @param {Object} data The data associated with the Resource.
 40 				 * @param 
 41 				 */
 42 				initialize: function($super, id, data)
 43 				{
 44 					$super(id);
 45 					
 46 					if(data != undefined) {
 47 						if(data.image_id != undefined) this.imageId = data.image_id;

 49 						
 50 						if(data.x != undefined) this.x = data.x;
 51 						if(data.y != undefined) this.y = data.y;

 53 						if(data.h != undefined) this.h = data.h;
 54 					}
 55 				},
 56 				
 57 				load: function(manager, callback)
 58 				{
 59 					this.imageResource = manager.getResource(
 60 						ImageResource.TYPE,
 61 						this.imageId
 62 					);
 63 					
 64 					this.img = this.imageResource.img;
 65 					
 66 					if(this.w == -1) this.w = this.img.width;
 67 					if(this.h == -1) this.h = this.img.height;
 68 					
 69 					this.loaded = true;
 70 					callback();
 71 				}
 72 			}
 73 		);
 74 		
 75 		SpriteResource.TYPE = 2;
 76 		
 77 		return SpriteResource;

 79 	}
