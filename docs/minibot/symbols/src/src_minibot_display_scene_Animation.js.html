1 define(
  2 	[
  3 		'minibot/display/scene/SceneDisplayObject'
  4 	],
  5 	function
  6 	(
  7 		SceneDisplayObject
  8 	)
  9 	{
 10 		var Animation = Class.create(
 11 			SceneDisplayObject,
 12 			/** @lends display.scene.Animation# */
 13 			{
 14 				
 15 				animation: null,
 16 				
 17 				currentFrame: 0,
 18 				
 19 				currentSprite: null,
 20 				
 21 				currentDelay: null,
 22 				
 23 				time: 0,
 24 				
 25 				loops: 0,
 26 				
 27 				playing: false,
 28 				
 29 				/**
 30 				 * Description of constructor.
 31 				 * @class Short description of class.
 32 				 * Long Description of class.
 33 				 * @extends display.scene.SceneDisplayObject
 34 				 * @constructs
 35 				 * @param {resource.AnimationResource} animation
 36 				 * @param {bool} play
 37 				 * @param {int} loops
 38 				 * @param
 39 				 */
 40 				initialize: function($super, animation, play, loops)
 41 				{
 42 					$super();
 43 					
 44 					this.loops = Animation.INFINITE_LOOPS;
 45 					
 46 					this.setAnimation(animation, play, loops);
 47 				},
 48 				
 49 				setAnimation: function(animation, play, loops)
 50 				{
 51 					this.playing = (play == undefined || play);
 52 					
 53 					if(loops == undefined) {
 54 						this.loops == Animation.INFINITE_LOOPS;
 55 					} else {
 56 						this.loops = loops;
 57 					}
 58 					
 59 					this.currentFrame = 0;
 60 					this.time = 0;
 61 					
 62 					this.animation = animation;
 63 					this.setupFrame();
 64 				},
 65 				
 66 				setupFrame: function()
 67 				{
 68 					if(this.animation == null) return;
 69 					this.currentSprite = this.animation.getSprite(this.currentFrame);
 70 					this.currentDelay = this.animation.getDelay(this.currentFrame);
 71 					if(this.w == 0) this.w = this.currentSprite.w;
 72 					if(this.h == 0) this.h = this.currentSprite.h;
 73 				},
 74 				
 75 				stop: function()
 76 				{
 77 					this.playing = false;
 78 					this.currentFrame = 0;
 79 					this.time = 0;
 80 					this.setupFrame();
 81 				},
 82 				
 83 				play: function(loops)
 84 				{
 85 					this.playing = true;
 86 					
 87 					if(loops == undefined) {
 88 						this.loops == Animation.INFINITE_LOOPS;
 89 					} else {
 90 						this.loops = loops;
 91 					}
 92 				},
 93 				
 94 				isPlaying: function()
 95 				{
 96 					return this.playing;
 97 				},
 98 				
 99 				render: function(dt, x, y)
100 				{
101 					if(this.animation == null) return;
102 					
103 					if(this.playing) {
104 						this.time += dt;
105 						while(this.time >= this.currentDelay) {
106 							
107 							// Handle looping
108 							if(this.loops != Animation.INFINITE_LOOPS) {
109 								if(this.animation.atEnd(this.currentFrame)) {
110 									if(this.loops == 0) {
111 										this.stop();
112 										this.setupFrame();
113 										break;
114 									} else {
115 										this.loops -= 1;
116 									}
117 								}
118 							}
119 							
120 							this.time -= this.currentDelay;
121 							this.currentFrame = this.animation.nextFrame(this.currentFrame);
122 							this.setupFrame();
123 						}
124 					}
125 					
126 					try {
127 						this.scene.drawImage(
128 							this.currentSprite.img,
129 							this.currentSprite.x, //sx,
130 							this.currentSprite.y, //sy,
131 							this.currentSprite.w, //sw,
132 							this.currentSprite.h, //sh,
133 							this.x + x, //dx,
134 							this.y + y, //dy,
135 							this.w, //dw,
136 							this.h //dh
137 						);
138 					} catch(error) {
139 						console.log('Animation: Rendering Fatal Error');
140 					}
141 					
142 				}
143 				
144 			}
145 		);
146 		
147 		Animation.INFINITE_LOOPS = -1;
148 		
149 		return Animation;
150 	}
