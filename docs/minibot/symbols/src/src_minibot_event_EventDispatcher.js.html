  7 define(
  8 	[],
  9 	function()
 10 	{
 11 		
 12 		var EventDispatcher = Class.create(
 13 			/** @lends event.EventDispatcher# */
 14 			{
 15 				/**
 16 				 * Map of eventListeners availiable.
 17 				 * @type json
 18 				 */
 19 				listeners: null,
 20 				
 21 				/**
 22 				 * Creates a new EventDispatcher instance.
 23 				 * @class Relegates and dispatches events to a target.
 24 				 * Events can be attached to targets in order to keep the flow of processes running as intended.
 25 				 * @constructs
 26 				 */
 27 				initialize: function()
 28 				{
 29 					this.listeners = {};
 30 				},
 31 				/**
 32 				 * Registers an eventListener to a target.
 33 				 * @param {type} type The data type of event the specified listener is looking for.
 34 				 * @param {object} callback The listener object that recieves the specified type.
 35 				 */
 36 				addEventListener: function(type, callback)
 37 				{
 38 					var callbacks;
 39 					if(this.listeners[type] == undefined) {
 40 						callbacks = new Array();
 41 						this.listeners[type] = callbacks;
 42 					} else {
 43 						callbacks = this.listeners[type];
 44 					}
 45 					callbacks.push(callback);
 46 				},
 47 				/**
 48 				 * Returns whether or not the EventDispatcher has an event listener for the specified type.
 49 				 * @param {type} type The data type of the event the specified listener is looking for.
 50 				 * @returns {boolean} A value indicating whether or not an event listener for the specified type is available.
 51 				 */
 52 				hasEventListener: function(type)
 53 				{
 54 					return (this.listeners[type] != undefined);
 55 				},
 56 				/**
 57 				 * Sends the event to the EventDispatcher to be sent into the event flow.
 58 				 * @param {event} event The event to be sent into the event flow.
 59 				 * @returns {boolean} A value indicating whether or not the event was successfully dispatched.
 60 				 */
 61 				dispatchEvent: function(event)
 62 				{
 63 					var callbacks;
 64 					if(this.listeners[event.type] == undefined) {
 65 						return false;
 66 					} else {
 67 						callbacks = this.listeners[event.type];
 68 					}
 69 					event.target = this;
 70 					for(var i = 0; i < callbacks.length; i++) {
 71 						var callback = callbacks[i];
 72 						callback(event);
 73 					}
 74 					return true;
 75 				},
 76 				/**
 77 				 * Gets rid of an eventListener.
 78 				 * @param {type} type The type of data the specified listener is looking for.
 79 				 * @param {object} callback The listener object that recieves the specified type.
 80 				 */
 81 				removeEventListener: function(type, callback)
 82 				{
 83 					var callbacks;
 84 					if(this.listeners[type] == undefined) {
 85 						return;
 86 					} else {
 87 						callbacks = this.listeners[type];
 88 					}
 89 					var i = callbacks.indexOf(callback);
 90 					if(i == -1) return;
 91 					callbacks.splice(i, 1);
 92 					
 93 					// Remove the callbacks array from the map if empty
 94 					if(callbacks.length == 0) {
 95 						delete this.listeners[type];
 96 					}
 97 				},
 98 				/**
 99 				 * Gets rid of all eventListeners on the map.
100 				 */
101 				removeAllEventListeners: function()
102 				{
103 					// TODO
104 				}
105 				
106 			}
107 		);
108 		
109 		return EventDispatcher;
110 		
111 	}
112 );
