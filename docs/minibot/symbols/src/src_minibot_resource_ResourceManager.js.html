  7 define(
  8 	[
  9 		'minibot/utils',
 10 		'minibot/core/Manager'
 11 	],
 12 	function
 13 	(
 14 		utils,
 15 		Manager
 16 	)
 17 	{
 18 		var ResourceManager = Class.create(
 19 			Manager,
 20 			/** @lends resource.ResourceManager# */
 21 			{
 22 				
 23 				typeOrder: null,
 24 				typeCount: null,
 25 				
 26 				typeMap: null,
 27 				
 28 				resourceMap: null,
 29 				
 30 				resourceCount: 0,
 31 				loadedCount: 0,
 32 				typeIndex: null,
 33 				typeLoadedCount: 0,
 34 				
 35 				/**
 36 				 * Description of constructor.
 37 				 * @class Short description of class.
 38 				 * Long Description of class.
 39 				 * @extends core.Manager
 40 				 * @constructs
 41 				 * @param {String} key The instance key.
 42 				 * @param 
 43 				 */
 44 				initialize: function($super, key)
 45 				{
 46 					$super(key);
 47 					this.typeOrder = [];
 48 					this.typeCount = [];
 49 					this.typeMap = {};
 50 					this.resourceMap = {};
 51 				},
 52 				
 53 				addType: function(type, className)
 54 				{
 55 					this.typeOrder.push(type);
 56 					this.typeCount.push(0);
 57 					this.typeMap[type] = className;
 58 					this.resourceMap[type] = {};
 59 				},
 60 				
 61 				addResource: function(type, id, data)
 62 				{
 63 					if(this.typeMap[type] == undefined) return;
 64 					var className = this.typeMap[type];
 65 					var resource = new className(id, data);
 66 					this.resourceMap[type][id] = resource;
 67 					this.resourceCount += 1;
 68 					
 69 					for(var i = 0; i < this.typeOrder.length; i++) {
 70 						if(type == this.typeOrder[i]) {
 71 							this.typeCount[i] += 1;
 72 							break;
 73 						}
 74 					}
 75 					
 76 				},
 77 				
 78 				loadAll: function(progressCallback, completeCallback)
 79 				{
 80 					this.progressCallback = progressCallback;
 81 					this.completeCallback = completeCallback;
 82 					
 83 					this.typeIndex = null;
 84 					this.loadNextType();
 85 					
 86 				},
 87 				
 88 				loadNextType: function()
 89 				{
 90 					if(this.typeIndex == null) {
 91 						this.typeIndex = 0;
 92 					} else {
 93 						this.typeIndex += 1;
 94 					}
 95 					
 96 					if(this.typeIndex >= this.typeOrder.length) {
 97 						this.completeCallback.bind(this).defer();
 98 						return;
 99 					}
100 					
101 					this.typeLoadedCount = 0;
102 					
103 					var type = this.typeOrder[this.typeIndex];
104 					var resources;
105 					var id;
106 					resources = this.resourceMap[type];
107 					for(id in resources) {
108 						this.loadResource(type, id);
109 					}
110 				},
111 				
112 				loadResource: function(type, id)
113 				{
114 					var resource = this.resourceMap[type][id];
115 					if(resource.isLoaded()) {
116 						this.handleResourceLoaded();
117 					} else {
118 						resource.load(this, this.handleResourceLoaded.bind(this));
119 					}
120 				},
121 				
122 				handleResourceLoaded: function()
123 				{
124 					this.typeLoadedCount += 1;
125 					var progress = Number(1 / this.resourceCount);
126 					this.progressCallback(progress);
127 					
128 					if(this.typeLoadedCount >= this.typeCount[this.typeIndex]) {
129 						this.loadNextType.bind(this).defer();
130 					}
131 				},
132 				
133 				getResource: function(type, id)
134 				{
135 					if(this.resourceMap[type] == undefined) return null;
136 					if(this.resourceMap[type][id] == undefined) return null;
137 					
138 					return this.resourceMap[type][id];
139 				}
140 				
141 			}
142 			
143 		);
144 		
145 		Object.extend(
146 			ResourceManager,
147 			/** @lends core.Manager */
148 			{
149 				getInstance: function(key)
150 				{
151 					if (!Manager.hasCore(key)) {
152 						new ResourceManager(key);
153 					}
154 					var retVal = Manager.getInstance(key);
155 					return retVal;
156 				}
157 			}
158 		);
159 		
160 		return ResourceManager;
161 		
162 	}
163 );
