1 define([
  2 	'minibot/utils',
  3 	'minibot/display/canvas/CanvasDisplayObject',
  4 	'minibot/event/UIEvent'
  5 ],
  6 function(
  7 	utils, 
  8 	CanvasDisplayObject,
  9 	UIEvent
 10 )
 11 {
 12 	return utils.define(
 13 		{
 14 			name: 'minibot.display.canvas.TextInput',
 15 			parent: CanvasDisplayObject
 16 		},
 17 		{
 18 			
 19 			input: null,
 20 			
 21 			inputX: null,
 22 			inputY: null,
 23 			inputW: null,
 24 			inputH: null,
 25 			
 26 			initialize: function($super, value, style)
 27 			{
 28 				$super();
 29 				
 30 				this.input = new Element('input', {type: 'text'});
 31 				
 32 				if(value != undefined) this.input.value = value;
 33 				if(style != undefined) this.input.setStyle(style);
 34 				
 35 				this.input.observe('focus', this.handleInputFocus.bindAsEventListener(this));
 36 				this.input.observe('blur', this.handleInputBlur.bindAsEventListener(this));
 37 			},
 38 			
 39 			setValue: function(value)
 40 			{
 41 				this.input.value = value;
 42 			},
 43 			
 44 			setType: function(type)
 45 			{
 46 				this.input.type = type;
 47 			},
 48 			
 49 			getValue: function()
 50 			{
 51 				return this.input.value;
 52 			},
 53 			
 54 			getType: function()
 55 			{
 56 				return this.input.type;
 57 			},
 58 			
 59 			onAddedToCanvas: function($super)
 60 			{
 61 				$super();
 62 				
 63 				console.log('added text');
 64 				this.canvas.addToHtmlOverlay(this.input);
 65 			},
 66 			
 67 			onRemovedFromCanvas: function($super)
 68 			{
 69 				$super();
 70 				
 71 				this.input.remove();
 72 			},
 73 			
 74 			render: function($super, dt, context, x, y)
 75 			{
 76 				
 77 				if(this.inputX != (this.x + x)) {
 78 					this.inputX = (this.x + x);
 79 					this.input.style.left = (this.inputX / this.canvas.scale) + "px";
 80 				}
 81 				
 82 				if(this.inputY != (this.y + y)) {
 83 					this.inputY = (this.y + y);
 84 					this.input.style.top = (this.inputY / this.canvas.scale) + "px";
 85 				}
 86 				
 87 				if(this.inputW != (this.w)) {
 88 					this.inputW = (this.w);
 89 					this.input.style.width = (this.inputW / this.canvas.scale) + "px";
 90 				}
 91 				
 92 				if(this.inputH != (this.h)) {
 93 					this.inputH = (this.h);
 94 					this.input.style.height = (this.inputH / this.canvas.scale) + "px";
 95 				}
 96 				
 97 				context.strokeRect(this.x + x, this.y + y, this.w, this.h);
 98 				
 99 				$super();
100 				
101 			},
102 			
103 			dispatchEvent: function($super, event)
104 			{
105 				console.log("TextInput::dispatchEvent");
106 				if(event.type == UIEvent.TOUCH_START) {
107 					this.input.focus();
108 				}
109 				return $super(event);
110 			},
111 			
112 			handleInputFocus: function(event)
113 			{
114 				var x = 0;
115 				var y = 0;
116 				
117 				var uIEvent = new UIEvent(UIEvent.FOCUS, {}, x, y);
118 				this.dispatchEvent(uIEvent);
119 			},
120 			
121 			handleInputBlur: function(event)
122 			{
123 				var x = 0;
124 				var y = 0;
125 				
126 				var uIEvent = new UIEvent(UIEvent.BLUR, {}, x, y);
127 				this.dispatchEvent(uIEvent);
128 			}
129 			
130 		}
131 	);
132 });
